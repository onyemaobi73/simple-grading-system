
START
  INPUT score
  IF score >= 70 AND score <= 100 THEN
    PRINT "Grade: A"
  ELSE IF score >= 60 THEN
    PRINT "Grade: B"
  ELSE IF score >= 50 THEN
    PRINT "Grade: C"
  ELSE IF score >= 45 THEN
    PRINT "Grade: D"
  ELSE IF score >= 40 THEN
    PRINT "Grade: E"
  ELSE
    PRINT "Grade: F"
END
